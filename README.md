# ws
workspace, manage multi task and update etc.

# useage 

```
# run ws, goto a shell;
# $xxx means you should input variable xxx;
status # get current status messae.
create task $task_name # create batch for task_name, if batch exist, get warn;
use task $task_name # change workspace to task's branch
push task  # push task to server.
update # update proj
build # build proj
finished # show finished task
delete $task_name # delete task's branches 
help # help message.




```
